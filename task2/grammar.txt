OP  -> (ATR | PRINT);

PRINT  -> print E

ATR  -> id { Print( lexema ); } = E { Print( "="); }

ARGS   -> ARGFST ARGSREM
ARGFST -> E
ARGSRE ->  , E 
         | , E ARGSREM
         | ε

E  -> T E'
    | id(ARGS)

E' -> + T { Print( "+"); } E'
    | - T { Print( "-"); } E'
    | ε
T -> F T'
T' -> * F { Print( "*"); } T'
    | / F { Print( "/"); } T'
    | ε
F -> id { Print(lexema + "@"); }
   | num { Print( lexema ); }
   | ( E )
